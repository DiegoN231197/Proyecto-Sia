CREATE TABLE Compone(cod_compone INT PRIMARY KEY,cantidad_mp INT, descripcion VARCHAR(100),cod_cerveza INT FOREING KEY, cod_materiaprima INT FOREING KEY);



CREATE TABLE Caducidad(cod_caducidad INT PRIMARY KEY, fecha VARCHAR(50), descripcion VARCHAR(100));



CREATE TABLE Materia prima(cod_materiaprima INT PRIMARY KEY, nombre VARCHAR(100), tipo VARCHAR(50), descripcion VARCHAR(100), cod_caducidad INT FOREING KEY);



CREATE TABLE Proveedor(cod_proveedor INT PRIMARY KEY, rut INT, direccion VARCHAR(100), cantidad INT, cod_materiaprima INT FOREING KEY);



CREATE TABLE Crea(cod_crea INT PRIMARY KEY, descripcion VARCHAR(100), cod_cerveza INT FOREING KEY, cod_empleado INT FOREING KEY, cod_tiempofabricacion INT FOREING KEY);



CREATE TABLE Tiempo fabricacion(cod_tiempofabricacion INT PRIMARY KEY, duracion VARCHAR(100), descripcion VARCHAR(100));
